25  package org.codehaus.mojo.animal_sniffer;
26  
27  import junit.framework.TestCase;
28  
29  public class SignatureCheckerTest extends TestCase
30  {
31  
32      public SignatureCheckerTest( String testName )
33      {
34          super( testName );
35      }
36  
37      public void testToSourceForm()
38      {
39          assertSourceForm( "java.util.HashMap", "java/util/HashMap", null );
40          assertSourceForm( "java.util.Map.Entry", "java/util/Map$Entry", null );
41          assertSourceForm( "String", "java/lang/String", null );
42          assertSourceForm( "java.lang.reflect.Field", "java/lang/reflect/Field", null );
43          assertSourceForm( "Thread.State", "java/lang/Thread$State", null );
44          assertSourceForm( "java.util.Set my.Class.myfield", "my/Class", "myfield#Ljava/util/Set;" );
45          assertSourceForm( "String[] my.Class.myfield", "my/Class", "myfield#[Ljava/lang/String;" );
46          assertSourceForm( "double[][][] my.Class.myfield", "my/Class", "myfield#[[[D" );
47          assertSourceForm( "void my.Class.mymethod()", "my/Class", "mymethod()V" );
48          assertSourceForm( "Object my.Class.mymethod(int, double, Thread)", "my/Class",
49                            "mymethod(IDLjava/lang/Thread;)Ljava/lang/Object;" );
50      }
51  
52      public void testToAnnotationDescriptor()
53      {
54          assertAnnotationDescriptor( "Lorg/codehaus/mojo/animal_sniffer/IgnoreJRERequirement;",
55                                      "org.codehaus.mojo.animal_sniffer.IgnoreJRERequirement" );
56          assertAnnotationDescriptor( "Lorg/jvnet/animal_sniffer/IgnoreJRERequirement;",
57                                      "org.jvnet.animal_sniffer.IgnoreJRERequirement" );
58  
59          assertAnnotationDescriptor( "Lcom/foo/Bar;", "com.foo.Bar" );
60          assertAnnotationDescriptor( "Lcom/foo/Bar;", "com/foo/Bar" );
61      }
62  
63      private static void assertSourceForm( String expected, String type, String sig )
64      {
65          assertEquals( expected, SignatureChecker.toSourceForm( type, sig ) );
66      }
67  
68      private static void assertAnnotationDescriptor( String expected, String fqn )
69      {
70          assertEquals( expected, SignatureChecker.toAnnotationDescriptor( fqn ) );
71      }
72  
73  }
